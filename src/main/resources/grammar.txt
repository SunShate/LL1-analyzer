<Program> -> <Logical_expr> || <Arithmetic_expr> || <Variable_declaration> || <Signature>

<Logical_expr> -> <Logical_variable> <R_logical_expr>
<Logical_variable> -> <Is_negative> <R_logical_variable>
<Is_negative> -> ! <D> || <D>
<D> -> ( <Logical_expr> ) || false || true
<R_logical_variable> -> && <Is_negative> <R_logical_variable> || eps
<R_logical_expr> -> || <Logical_variable> <R_logical_expr> || eps

<Arithmetic_expr> -> <Term> <R_arithmetic_expr>
<R_arithmetic_expr> -> + <Arithmetic_expr> || - <Arithmetic_expr> || eps
<Term> -> <Factor> <R_term>
<R_term> -> * <Term> || / <Term> || eps
<Factor> -> number || ( <Arithmetic_expr> ) || - <Factor>

<Variable_declaration> -> <Type> <Identifier> <Value> ;
<Value> -> <Identifier> || <Integer_const> || <Boolean_const>

<Identifier> -> String || eps

<Integer_const> -> <Minus> <Number>
<Minus> -> - <Non_zero_digit> || eps
<Number> -> <Digit> <Number> || eps
<Digit> -> <Non_zero_digit> || 0
<Non_zero_digit> -> 1 || ... || 9

<Boolean_const> -> true || false

<Type> -> void || int || boolean || double

<Signature> -> <Type> <Identifier> ( <Signature_end>
<Signature_end> -> ) ; || <Param> <Param_list> ) ;
<Param> -> <Type> <Identifier> || ...
<Param_list> -> , <Param> <Param_list> || eps
